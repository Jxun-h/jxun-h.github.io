n, k = map(int, input().split())


def f(x):
    return (x + 1) * (n - x + 1)


lo, hi = 0, n // 2 + 1
while lo != hi:

    mid = (lo + hi) // 2
    data = f(mid)

    if data == k:
        print("YES")

    if data > k:
        hi = mid

    else:
        lo = mid + 1
