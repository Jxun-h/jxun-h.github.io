n = int(stdin.readline())

    nums.append(length)

        x.append(length)

    x = abs(nums[xi - 1] - nums[0])

    x = abs(nums[xi - 1] - nums[xi + 1])

if yi + 1 >= 6:
    y = abs(nums[yi - 1] - nums[0])
else:
    y = abs(nums[yi - 1] - nums[yi + 1])

print(((max_x * max_y) - (x * y)) * n)
