from collections import deque
from sys import stdin, setrecursionlimit

setrecursionlimit(10 ** 6)


def get_cnt(x, graph, visited):
    q = deque()
    q.append(x)
    visited.add(x)

    while q:
        now = q.popleft()

        for y in list(graph[now]):
            if y not in visited:
                q.append(y)
                visited.add(y)

    return 1
