from sys import stdin, setrecursionlimit
from collections import deque

setrecursionlimit(int(1e9))

n = int(stdin.readline())
arr = [list(map(int, stdin.readline().split())) for _ in range(n)]
answer, q = 0, deque()


def get(i, j):
    if arr[i][j]:  
        q.append(arr[i][j])  
        arr[i][j] = 0  


def merge(i, j, di, dj):  
    while q:
        x = q.popleft()  
        if not arr[i][j]:  
            arr[i][j] = x
        elif arr[i][j] == x:  
            arr[i][j] = x * 2  
            i, j = i + di, j + dj
        else:  
            i, j = i + di, j + dj
            arr[i][j] = x


def move(k):

    if k == 0:  
        for j in range(n):
            for i in range(n):
                get(i, j)
            merge(0, j, 1, 0)  
    elif k == 1:  
        for j in range(n):
            for i in range(n - 1, -1, -1):
                get(i, j)
            merge(n - 1, j, -1, 0)  
    elif k == 2:  
        for i in range(n):
            for j in range(n):
                get(i, j)
            merge(i, 0, 0, 1)  
    else:  
        for i in range(n):
            for j in range(n - 1, -1, -1):
                get(i, j)
            merge(i, n - 1, 0, -1)  


def recursive(count):
    global arr, answer
    if count == 5:  
        for i in range(n):
            answer = max(answer, max(arr[i]))  
        return
    b = [x[:] for x in arr]  

    for k in range(4):  
        move(k)  
        recursive(count + 1)  
        arr = [x[:] for x in b]


recursive(0)
print(answer)
