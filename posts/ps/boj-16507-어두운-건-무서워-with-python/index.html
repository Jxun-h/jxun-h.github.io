    data = list(map(int, stdin.readline().split()))

    addr = [(a, b), (c, d)]
    addr.sort(key=lambda x: x[0])
    res = 0

    knife = (abs(addr[0][0]-addr[1][0]) + 1) * (abs(addr[0][1]-addr[1][1]) + 1)
    for i in range(addr[0][0], addr[1][0] + 1):

    print(res // knife)
