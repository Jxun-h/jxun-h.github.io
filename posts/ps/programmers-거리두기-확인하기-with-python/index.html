from collections import deque

dx, dy = [1, -1, 0, 0], [0, 0, -1, 1]


def bfs(board, now, goal, dist):
    q = deque()
    visited = []
    x, y = now
    q.append((0, x, y, []))
    visited.append(now)
    check = False

    while q:

                    check = True
                elif board[i][j] == 'O':
                    return False

        for i in range(4):

def solution(places):
    answer = []

    for i in range(len(places)):
        board = []
        data = places[i]

        ps = []

        # ëŒ€ê¸°ì‹¤ ë¦¬ìŠ¤íŠ¸ ë§Œë“¤ê¸°
        for j in range(5):

                if data[j][k] == 'P':
                    ps.append((j, k))
            board.append(list(data[j]))

        ps.sort()
        check = True
        for j in range(len(ps)):
            x1, y1 = ps[j]
            for k in range(j + 1, len(ps)):
                x2, y2 = ps[k]
                dist = abs(x1 - x2) + abs(y1 - y2)
                if dist < 3:
                    if not bfs(board, ps[j], ps[k], dist):

            if not check:

        answer.append(1) if check else answer.append(0)

    return answer
